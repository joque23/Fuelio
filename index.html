```
Pre prepojenie s Google Sheets potrebuješ Google Apps Script – bezplatná služba od Google. Tvoje dáta sa uložia priamo do tvojho Google Sheetu.

1Otvor script.google.com a vytvor nový projekt.
2Vlož skript nižšie do editora a klikni Nasadiť → Nové nasadenie → Web App. Prístup nastav na Ktokoľvek.
3Skopíruj vygenerovanú URL a vlož ju sem.
Google Apps Script kód (skopíruj):
function doPost(e) {
```

try {
var data = JSON.parse(e.postData.contents);
var ss = SpreadsheetApp.openById(data.sheetId);
var sheet = ss.getSheetByName(data.sheetName)
|| ss.insertSheet(data.sheetName);
// Add header row if sheet is empty
if (sheet.getLastRow() === 0) {
sheet.appendRow(data.headers);
sheet.getRange(1, 1, 1, data.headers.length)
.setFontWeight(‘bold’)
.setBackground(’#1a3520’)
.setFontColor(’#4ade80’);
sheet.setFrozenRows(1);
}
// Check for duplicate by ID (column A)
var idCol = sheet.getRange(‘A:A’).getValues().flat();
var existingRow = idCol.indexOf(data.row[0]);
if (existingRow > 0) {
// Update existing row
sheet.getRange(existingRow + 1, 1, 1, data.row.length).setValues([data.row]);
} else {
sheet.appendRow(data.row);
}
return ContentService.createTextOutput(
JSON.stringify({ok: true})

} catch(err) {
return ContentService.createTextOutput(
JSON.stringify({ok: false, error: err.toString()})
).setMimeType(ContentService.MimeType.JSON);
}
}

function doGet(e) {
try {
var ss = SpreadsheetApp.openById(e.parameter.sheetId);
var sheet = ss.getSheetByName(e.parameter.sheetName);
if (!sheet) return ContentService.createTextOutput(
JSON.stringify({rows:[]})
).setMimeType(ContentService.MimeType.JSON);
var data = sheet.getDataRange().getValues();
return ContentService.createTextOutput(
JSON.stringify({rows: data})
).setMimeType(ContentService.MimeType.JSON);
} catch(err) {
return ContentService.createTextOutput(
JSON.stringify({rows:[], error: err.toString()})
).setMimeType(ContentService.MimeType.JSON);
}
}
