Google Sheets prepojenie

Pre prepojenie s Google Sheets potrebuješ Google Apps Script. Tvoje dáta sa uložia priamo do tvojho Google Sheetu.

1Otvor script.google.com a vytvor nový projekt.
2Vlož kód nižšie, nasaď ako Web App (prístup: Ktokoľvek).
3Skopíruj URL a vlož ju do poľa nižšie.
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(data.sheetId);
    var sheet = ss.getSheetByName(data.sheetName) || ss.insertSheet(data.sheetName);
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(data.headers);
      sheet.getRange(1,1,1,data.headers.length).setFontWeight('bold');
      sheet.setFrozenRows(1);
    }
    var idCol = sheet.getRange('A:A').getValues().flat();
    var existingRow = idCol.indexOf(data.row[0]);
    if (existingRow > 0) {
      sheet.getRange(existingRow+1,1,1,data.row.length).setValues([data.row]);
    } else { sheet.appendRow(data.row); }
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
function doGet(e) {
  try {
    var ss = SpreadsheetApp.openById(e.parameter.sheetId);
    var sheet = ss.getSheetByName(e.parameter.sheetName);
    if (!sheet) return ContentService.createTextOutput(JSON.stringify({rows:[]})).setMimeType(ContentService.MimeType.JSON);
    return ContentService.createTextOutput(JSON.stringify({rows:sheet.getDataRange().getValues()})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({rows:[],error:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
