Google Sheets prepojenie

Pre prepojenie potrebuješ Google Apps Script. Dáta sa uložia priamo do tvojho Google Sheetu.

1Otvor script.google.com a vytvor nový projekt.
2Vlož kód nižšie → Nasadiť → Web App → prístup: Ktokoľvek.
3Skopíruj URL a vlož ju do poľa nižšie.
function doPost(e){
  try{
    var d=JSON.parse(e.postData.contents);
    var ss=SpreadsheetApp.openById(d.sheetId);
    var sh=ss.getSheetByName(d.sheetName)||ss.insertSheet(d.sheetName);
    if(sh.getLastRow()===0){sh.appendRow(d.headers);sh.setFrozenRows(1);}
    var ids=sh.getRange('A:A').getValues().flat();
    var row=ids.indexOf(d.row[0]);
    if(row>0)sh.getRange(row+1,1,1,d.row.length).setValues([d.row]);
    else sh.appendRow(d.row);
    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  }catch(e){return ContentService.createTextOutput(JSON.stringify({ok:false,error:e.toString()})).setMimeType(ContentService.MimeType.JSON);}
}
function doGet(e){
  try{
    var ss=SpreadsheetApp.openById(e.parameter.sheetId);
    var sh=ss.getSheetByName(e.parameter.sheetName);
    if(!sh)return ContentService.createTextOutput(JSON.stringify({rows:[]})).setMimeType(ContentService.MimeType.JSON);
    return ContentService.createTextOutput(JSON.stringify({rows:sh.getDataRange().getValues()})).setMimeType(ContentService.MimeType.JSON);
  }catch(e){return ContentService.createTextOutput(JSON.stringify({rows:[],error:e.toString()})).setMimeType(ContentService.MimeType.JSON);}
}
